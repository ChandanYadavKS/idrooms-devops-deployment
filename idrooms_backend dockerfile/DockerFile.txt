FROM python:3.11-slim

ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1

WORKDIR /app

RUN apt-get update && apt-get install -y \
    build-essential \
    libffi-dev \
    libpango-1.0-0 \
    libcairo2 \
    libgdk-pixbuf-2.0-0 \
    libgdk-pixbuf2.0-bin \
    libgobject-2.0-0 \
    libglib2.0-0 \
    libpangocairo-1.0-0 \
    libxml2 \
    libxslt1.1 \
    libharfbuzz0b \
    libfribidi0 \
    fontconfig \
    fonts-dejavu \
    fonts-liberation \
    libjpeg62-turbo \
    libpng16-16 \
    default-libmysqlclient-dev \
    netcat-traditional \
    ca-certificates \
    curl \
    && rm -rf /var/lib/apt/lists/*

COPY requirements.txt .
RUN pip install --upgrade pip && pip install --no-cache-dir -r requirements.txt

COPY . .

# Create non-root user
RUN useradd -m appuser
USER appuser

CMD ["gunicorn", "idrooms.wsgi:application", "--bind", "0.0.0.0:8000"]
