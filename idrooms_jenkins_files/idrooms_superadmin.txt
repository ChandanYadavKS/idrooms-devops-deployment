pipeline {
    agent any

    environment {
        IMAGE_NAME     = "superadmin_idrooms"
        CONTAINER_NAME = "superadmin_idrooms"
        HOST_PORT      = "3004"
        CONTAINER_PORT = "3004"

        REPO_URL       = "https://github.com/akmusic-mad/idrooms_superadmin_panel.git"
        ENV_CRED_ID    = "marketing_env"
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'production',
                    url: "${REPO_URL}",
                    credentialsId: 'github-private'
            }
        }

        stage('Setup .env') {
            steps {
                withCredentials([file(credentialsId: "${ENV_CRED_ID}", variable: 'ENV_FILE')]) {
                    sh '''
                        chmod -R 777 $WORKSPACE
                        cp "$ENV_FILE" .env
                    '''
                }
            }
        }

        stage('Build Docker Image') {
            steps {
                sh '''
                    echo "🔨 Building Docker image"
                    docker build -t ${IMAGE_NAME} .
                '''
            }
        }

        stage('Stop Previous Container') {
            steps {
                sh '''
                    echo "🛑 Stopping old container (if exists)"
                    docker rm -f ${CONTAINER_NAME} || true
                '''
            }
        }

        stage('Run New Container') {
            steps {
                sh '''
                    echo "▶️ Running new container"
                    docker run -d \
                        --env-file .env \
                        -p ${HOST_PORT}:${CONTAINER_PORT} \
                        --name ${CONTAINER_NAME} \
                        ${IMAGE_NAME}
                '''
            }
        }

        stage('Cleanup Images') {
            steps {
                sh '''
                    echo "🧹 Cleaning unused images"
                    docker image prune -f
                '''
            }
        }
    }

    post {
        success {
            echo "✅ Superadmin deployed locally on port ${HOST_PORT}"
        }
        failure {
            echo "❌ Superadmin deployment failed"
        }
    }
}
